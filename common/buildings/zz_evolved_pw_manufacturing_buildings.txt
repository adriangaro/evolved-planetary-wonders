pw_building_mantle_crucible = {
	base_buildtime = @pw_building_buildtime
	position_priority = 120
	category = manufacturing
	base_cap_amount = 1
	can_be_ruined = no

	prerequisites = {
		pw_tech_planetary_mantle_production
	}

	potential = {
		exists = owner
		NOT = { has_modifier = resort_colony }
		NOT = { habitable_structure = yes }
	}

	allow = {
		has_major_upgraded_capital = yes
		OR = {
			free_district_slots > 0 # Needs a free district to occupy

			trigger = {
				has_modifier = pw_mod_mantle_crucible_alloys
			}
			volatile_motes = @pw_building_upkeep_rare_1
		}
		upkeep = {
			trigger = {
				has_modifier = pw_mod_mantle_crucible_consumer_goods
			}
			rare_crystals = @pw_building_upkeep_rare_1
		}
		upkeep = {
			trigger = {
				exists = owner
				owner = { country_uses_consumer_goods = no }
			}
			volatile_motes = @pw_building_upkeep_rare_1
		}
		inline_script = buildings/resources/inline_tec_pw_wonder_extra
		inline_script = buildings/resources/inline_tec_pw_wonders_beyond_ambition
		inline_script = buildings/resources/inline_tec_pw_monumentality_architecture_parlante

		
	}

	destroy_trigger = {
		OR = {
			has_modifier = resort_colony
			habitable_structure = yes
		}
	}

	convert_to = {
		building_factory_2
		building_foundry_2
	}

	upgrades = {
		pw_building_titan_forge
		pw_building_industrial_hearth
	}

	#Reduces district slots without modifier
	inline_script =  buildings/effects/inline_tec_pw_mod_integrated_monuments

	#Decrease Habitability
	triggered_planet_modifier = {
		potential = { #If habitability is not fixed
			exists = owner
			is_ideal_planet_class = {
				who = owner
				status = no
			}
			owner = { NOT = { has_country_flag = pw_mantle_crucible_habitability_project_completed }}
		}
		pop_environment_tolerance = -0.20 #Decrease Habitability
	}

	#Alloys
	inline_script = {
		script = jobs/inline_evolved_foundry
		count = 4
		mult = 1
		trigger = ""
		display = with_desc
	}
	

	#Consumer Goods
	inline_script = {
		script = jobs/inline_evolved_artisan
		count = 4
		mult = 1
		trigger = "
			owner = { 
				country_uses_consumer_goods = yes
			} 
		"
		display = with_desc
	}

	#Alloys (no CG)
	inline_script = {
		script = jobs/inline_evolved_foundry
		count = 2
		mult = 1
		trigger = "
			owner = { 
				country_uses_consumer_goods = no
			} 
		"
		display = no_desc
	}
	inline_script = {
		script = jobs/inline_evolved_miner

		hive_aquatic_count = 2

		trigger = "
			owner = { 
				country_uses_consumer_goods = no
			} 
		"

	#Ambition Bonuses
	triggered_planet_modifier = {

				has_edict = pw_wonders_beyond_ambition

		}
	}

		potential = {
			exists = owner

				country_uses_consumer_goods = yes
				has_edict = pw_wonders_beyond_ambition

			NOT = { has_active_building = pw_building_mantle_crucible }
			NOT = { has_building_construction = pw_building_mantle_crucible }
		}
		modifier = {
			factor = 1.5
			exists = owner
			OR = {
				owner = { is_machine_empire = yes }
				owner = { is_lithoid_empire = yes }
			}
		}
		modifier = {
			factor = 3
			OR = {
				has_active_building = building_foundry_1
				has_active_building = building_foundry_2
				has_active_building = building_foundry_3

				has_active_building = building_factory_1
				has_active_building = building_factory_2
				has_active_building = building_factory_3
			}
		}
		modifier = {
			factor = 5
			OR = {
				has_foundry_designation = yes
				has_factory_designation = yes
				has_industrial_designation = yes
			}
		}
		modifier = {
			factor = 2
			exists = owner
			owner = {
				has_monthly_income = {
					resource = alloys
					value < 100
				}
			}
		}
		modifier = {
			factor = 2
			exists = owner
			owner = {
				has_monthly_income = {
					resource = consumer_goods
					value < 10
				}
			}
		}
		modifier = {
			factor = 2
			planet.owner = {
				is_specialist_subject_type = { TYPE = prospectorium }
			}
		}
	}
}
